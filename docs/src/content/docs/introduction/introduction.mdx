### Installation

Install vue-fluid-dnd using your preferred package manager:

```bash
# with npm:
npm i vue-fluid-dnd

# with yarn:
yarn add vue-fluid-dnd

# with pnpm:
pnpm i vue-fluid-dnd
```

### Usage

#### Usage on global

Setup components on the main file:

export const globalUsage = `
import { createApp } from "vue";
import App from './App.vue';

import { Draggable, Droppable } from 'vue-fluid-dnd';
import "vue3-fluid-dnd/style.css";

const app = createApp(App);

app.component('Droppable', Droppable);
app.component('Draggable', Draggable);

app.mount('#app');`;

<Code code={globalUsage} lang="js" />

#### Usage on local

Import components on .vue files and styles:

export const optionApiCode = `
<script>
  import { Draggable, Droppable } from 'vue-fluid-dnd';
  import "vue3-fluid-dnd/style.css";

export default {
components: { Draggable, Droppable },
}

</script>`;
export const compositionApiCode = `
<script>
  import { Draggable, Droppable } from 'vue3-fluid-dnd';
  import "vue-fluid-dnd/style.css";

</script>`;
